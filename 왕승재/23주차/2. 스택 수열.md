#### 2. 스택 수열 (S2)

```cpp
#include <iostream>
#include <deque>

using namespace std;

int main() {
  ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);

  int n;
  cin >> n;

  deque<int> nums;

  for (int i = 1; i <= n; i++) {
    int num;
    cin >> num;

    nums.push_front(num);
  }

  deque<int> stack;
  deque<char> answer;

  int count = 1;

  while (!nums.empty()) {
    int num = nums.back();

    if (!stack.empty() && stack.back() == num) {
      stack.pop_back();
      answer.push_back('-');
      nums.pop_back();
    } else {
      stack.push_back(count);
      answer.push_back('+');
      count++;
    }

    if (count > 100001) {
      cout << "NO";
      return 0;
    }
  }

  for (int i = 0; i < answer.size(); i++) {
    cout << answer[i] << '\n';
  }
}
```
