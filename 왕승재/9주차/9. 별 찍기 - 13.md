#### 9. 별 찍기 - 13 (B3)

```cpp
#include <iostream>

using namespace std;

int main() {
  int n;
  
  cin >> n;
  
  for (int i = 0; i < n; i++) {
    for (int k = 0; k <= i; k++) {
      cout << '*';
    }
    
    cout << '\n';
  }
  
  for (int i = 0; i < n; i++) {
    for (int k = n-i-1; k > 0; k--) {
      cout << '*';
    }

    cout << '\n';
  }
}
```
