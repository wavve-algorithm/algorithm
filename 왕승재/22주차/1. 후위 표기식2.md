#### 1. 후위 표기식2 (S3)

```cpp
#include <iostream>
#include <vector>
#include <string>
#include <iomanip>

using namespace std;

int main() {
  ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);

  int n;
  cin >> n;

  string expression;
  cin >> expression;
  cin.ignore();

  vector<double> values(n);
  for (int i = 0; i < n; i++) {
    cin >> values[i];
  }

  vector<double> stack;
  for (char c: expression) {
    if(isalpha(c)) {
      stack.push_back(values[c - 'A']);
    } else {
      double b = stack.back();
      stack.pop_back();
      double a = stack.back();
      stack.pop_back();

      switch(c) {
        case '+':
          stack.push_back(a + b);
          break;
        case '-':
          stack.push_back(a - b);
          break;
        case '*':
          stack.push_back(a * b);
          break;
        case '/':
          stack.push_back(a / b);
          break;
      }
    }
  }

  cout << fixed << setprecision(2) << stack.back();
}
```
