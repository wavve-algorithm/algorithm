#### 1. 진법 변환

```cpp
#include <iostream>
#include <string>
#include <cmath>

using namespace std;

int main() {
  ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);

  string n;
  int b;

  cin >> n >> b;
  
  int count = 0, answer = 0;
  
  for (char &i: n) {
    answer += pow(b, count) * ((int)i - 55);
    count += 1;
  }

  cout << answer;
}
```
