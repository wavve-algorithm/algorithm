from collections import defaultdict, deque

def solution(n, path, order): 
    # create tree
    tree = defaultdict(set)
    for a, b in path:
        tree[b].add(a)
        tree[a].add(b)
    
    # order_dict
    # todo: ê°œì„  ê°€ëŠ¥í•  ê²ƒìœ¼ë¡œ ë³´ì„
    order_dict = {}
    before = order_dict.keys()
    blocked = [False for i in range(n)]
    for a, b in order:
        order_dict[a]=b
        blocked[b]=True
        
    # bfs
    visited = [False for i in range(n)]
    pending = [False for i in range(n)]
    dq = deque([0])

    while(dq):
        cur = dq.popleft()
        if(blocked[cur]):
            pending[cur]=True
            continue
        
        visited[cur]=True
        if(cur in before):
            if(pending[order_dict[cur]]):
                dq.append(order_dict[cur])
                pending[order_dict[cur]]=False
            blocked[order_dict[cur]]=False
            
        for nxt in tree[cur]:
            if(not visited[nxt]):
                dq.append(nxt)

    if(True in pending): return False

    return True
