```python
from collections import deque

def solution(priorities, location):
    answer=0
    q = deque(enumerate(priorities))
    
    def hasHighPriorityProcess(priority):
        for p in q:
            if(p[1]>priority):
                return True
        return False
    
    while(q):
        [index, value] = cur = q.popleft()
        if(hasHighPriorityProcess(value)):
            q.append(cur)
        else:
            answer+=1
            if(index==location):
                return answer
        
    return -1
```
