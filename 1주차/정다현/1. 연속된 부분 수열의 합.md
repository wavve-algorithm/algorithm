def solution(sequence, k):  
 candidates = []

    l = len(sequence)

    currentResult = []
    currentLength = float('inf')

    start = 0
    end = 0

    sum = 0
    while (end < l and start <= end):
        while (sum <= k):
            sum += sequence[end]
            end += 1
        if (sum == k and (end - start < currentLength)):
            currentResult = [start, end]
            currentLength = end - start
        sum = sum - sequence[start]
        start = start + 1
        end = max(start, end)

    return currentResult
